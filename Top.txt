-- 1. Таблица для счетчика посещений
CREATE TABLE visits (
  id BIGSERIAL PRIMARY KEY,
  visitor_id TEXT NOT NULL,
  visited_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Индексы для быстрого поиска
CREATE INDEX idx_visits_visitor_id ON visits(visitor_id);
CREATE INDEX idx_visits_visited_at ON visits(visited_at);

-- Политики доступа (включите в Supabase)
-- Разрешить анонимный INSERT
CREATE POLICY "Allow anonymous insert" ON visits
  FOR INSERT WITH CHECK (true);

-- Разрешить анонимный SELECT  
CREATE POLICY "Allow anonymous select" ON visits
  FOR SELECT USING (true);

-- 2. Таблица для Telegram статуса
CREATE TABLE telegram_status (
  id BIGSERIAL PRIMARY KEY,
  user_id TEXT NOT NULL UNIQUE,
  is_online BOOLEAN DEFAULT false,
  last_seen TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Индексы
CREATE INDEX idx_telegram_status_user_id ON telegram_status(user_id);
CREATE INDEX idx_telegram_status_updated_at ON telegram_status(updated_at);

-- Политики доступа
CREATE POLICY "Allow anonymous insert telegram" ON telegram_status
  FOR INSERT WITH CHECK (true);

CREATE POLICY "Allow anonymous update telegram" ON telegram_status
  FOR UPDATE USING (true);

CREATE POLICY "Allow anonymous select telegram" ON telegram_status
  FOR SELECT USING (true);
